\centerline{\bf Mátrix-mátrix }
\noindent Egy $k\times m$-es $A=(a_{ij})$ mátrix és egy $m\times k$-s $B=(b_{ij})$ mátrix szorzatának kiszámítása.


\noindent
{\bf Input}
\begin{flalign*}
& k\: m\: n &\\
& a_{11}\ldots a_{1 m}\\
& \ldots \\
& a_{k 1}\ldots a_{k m}\\
& b_{11}\ldots a_{1 n}\\
& \ldots \\
& b_{m 1}\ldots b_{m n}\\
\end{flalign*}


\noindent
{\bf Output}
\begin{flalign*}
& AB &
\end{flalign*}


\noindent
{\bf Korlátok}\newline
$0<m,n<100.$ 
A kiírt számok {\bf 12} értékes jegyet tartalmazzanak!
